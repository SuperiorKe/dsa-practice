Payment Workflow with Stripe Webhooks

// Secure webhook handler for payment confirmation async function handleStripeWebhook(request) { const signature = request.headers['stripe-signature']; // Verify webhook signature for security const event = stripe.webhooks.constructEvent( request.body, signature, process.env.STRIPE_WEBHOOK_SECRET ); if (event.type === 'payment_intent.succeeded') { // Update order status in database await updateOrderStatus(event.data.object.id, 'paid'); // Trigger confirmation email via SendGrid await sendOrderConfirmation(event.data.object.metadata); } return { received: true }; }

Database Schema for Transaction Records

-- PostgreSQL schema with ACID compliance CREATE TABLE orders ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), customer_id UUID REFERENCES customers(id), stripe_payment_id VARCHAR(255) UNIQUE NOT NULL, total_amount DECIMAL(10,2) NOT NULL, status VARCHAR(50) DEFAULT 'pending', created_at TIMESTAMP DEFAULT NOW(), updated_at TIMESTAMP DEFAULT NOW() ); -- Index for fast payment lookups CREATE INDEX idx_stripe_payment ON orders(stripe_payment_id);
